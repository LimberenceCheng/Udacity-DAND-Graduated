Create a new function to transform the carat variable

cuberoot_trans = function(){
  trans_new('cuberoot',
            transform = function(x) {x^(1/3)},
            inverse = function(x){ x^3})
  }

Building the Linear Model

m2 <- update(m1, ~ . + carat)
m3 <- update(m2, ~ . + cut)
m4 <- update(m3, ~ . + color)
m5 <- update(m4, ~ . + clarity)
